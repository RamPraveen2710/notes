Overall Delivery Score = 
VAR TotalResponses = COUNT('TableName'[Overall Delivery of Solutions])
VAR PositiveResponses = CALCULATE(
    COUNT('TableName'[Overall Delivery of Solutions]), 
    'TableName'[Overall Delivery of Solutions] IN {"Agree", "Strongly Agree"}
)
RETURN DIVIDE(PositiveResponses, TotalResponses, 0) * 100


Satisfied Employees Percentage = 
VAR SatisfiedCount = 
    CALCULATE(
        COUNT(employee_pro_bono_survey[Overall Experience with the Program]), 
        employee_pro_bono_survey[Satisfaction Score] >= 4
    )
VAR TotalResponses = COUNT(employee_pro_bono_survey[Overall Experience with the Program])

RETURN DIVIDE(SatisfiedCount, TotalResponses, 0) * 100
